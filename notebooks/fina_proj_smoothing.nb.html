df_mon <- weather_data %>%
  mutate(weather_data, yearmon = str_sub(weather_data[,'datetime'], start = 1, end = 7) ) %>%
  group_by(yearmon)　%>%
  mutate(avgMTemp = mean(temp)) %>% 
  distinct(yearmon,.keep_all = TRUE) %>%
  select(c(yearmon, avgMTemp))

df_mon_train <- filter(df_mon, yearmon < "2021-06")
